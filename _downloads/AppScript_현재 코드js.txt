const SHEET_NAME = 'data';

// =====================================================================
// 이 윗부분에 있는 사용자님의 기존 SPREADSHEET_ID 와 SHEET_NAME 은 그대로 두세요!
// 아래의 내용만 복사해서 기존 코드의 나머지 부분을 덮어쓰기 하시면 됩니다.
// =====================================================================

// 1. CORS Preflight 요청 처리 (Vercel 연동을 위한 필수 추가)
function doOptions(e) {
  return ContentService.createTextOutput("")
    .setMimeType(ContentService.MimeType.TEXT)
    .setHeader("Access-Control-Allow-Origin", "*")
    .setHeader("Access-Control-Allow-Methods", "GET, POST, OPTIONS")
    .setHeader("Access-Control-Allow-Headers", "Content-Type");
}

// 2. GET 요청 처리 (데이터 불러오기 - 개선됨)
function doGet(e) {
  try {
    const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(SHEET_NAME);
    const data = sheet.getDataRange().getValues();

    const headers = data[0];
    const rows = data.slice(1);
    const result = rows.map(row => {
      let obj = {};
      headers.forEach((header, index) => {
        obj[header] = row[index];
      });
      return obj;
    });

    return buildResponse({ success: true, data: result });
  } catch (error) {
    return buildResponse({ success: false, error: error.toString() });
  }
}

// 3. POST 요청 처리 (현재 코드의 방식 유지 + 일괄 저장 기능 추가)
function doPost(e) {
  try {
    const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(SHEET_NAME);
    const body = JSON.parse(e.postData.contents);
    
    // 배열(여러 개의 데이터)이 들어올 경우 한 번에 저장
    if (Array.isArray(body)) {
      const rowsToAppend = body.map(item => [
        item.date,
        item.ticker,
        item.shares,
        item.averageCost
      ]);
      
      const lastRow = sheet.getLastRow();
      if (rowsToAppend.length > 0) {
        sheet.getRange(lastRow + 1, 1, rowsToAppend.length, 4).setValues(rowsToAppend);
      }
    } else {
      // 기존처럼 1개의 데이터만 들어올 경우 (현재 코드 로직 그대로 유지)
      const rowData = [
        body.date,
        body.ticker,
        body.shares,
        body.averageCost
      ];
      sheet.appendRow(rowData);
    }
    
    return buildResponse({ success: true, message: "Data saved successfully" });
  } catch (error) {
    return buildResponse({ success: false, error: error.toString() });
  }
}

// 4. 공통 응답 처리 함수 (JSON 응답 및 CORS 헤더 강제 적용 - 추가됨)
function buildResponse(responseObject) {
  return ContentService.createTextOutput(JSON.stringify(responseObject))
    .setMimeType(ContentService.MimeType.JSON)
    .setHeader("Access-Control-Allow-Origin", "*")
    .setHeader("Access-Control-Allow-Methods", "GET, POST, OPTIONS")
    .setHeader("Access-Control-Allow-Headers", "Content-Type");
}
